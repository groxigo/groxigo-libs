Usage Guide

How to use tokens in your projects.

CSS Usage

/* Import in your CSS file */
@import '@groxigo/tokens/dist/css/tokens.css';

/* Or link in HTML */
<link rel="stylesheet" href="node_modules/@groxigo/tokens/dist/css/tokens.css">
/* Primitive colors */
var(--color-blue-600)
var(--color-gray-100)

/* Semantic tokens */
var(--surface-primary)
var(--text-primary)
var(--brand-primary)
var(--status-success)

/* Component tokens */
var(--button-primary-bg)
var(--input-border)
var(--card-bg)
.button {
  background: var(--button-primary-bg);
  color: var(--button-primary-text);
  border-radius: var(--radius);
  padding: var(--spacing-2) var(--spacing-4);
  transition: var(--transition-button);
}

.button:hover {
  background: var(--button-primary-bg-hover);
}

.card {
  background: var(--card-bg);
  border: 1px solid var(--card-border);
  border-radius: var(--radius);
  box-shadow: var(--shadow-sm);
}

React Native Usage

import { tokens, getThemeColors } from '@groxigo/tokens';
import { tokens } from '@groxigo/tokens';

const styles = StyleSheet.create({
  button: {
    backgroundColor: tokens.colors.semantic.brand.primary.default,
    padding: tokens.spacing[4],
    borderRadius: tokens.radius.md,
  },
  text: {
    color: tokens.colors.semantic.text.primary,
    fontSize: tokens.typography.fontSize.base,
  },
});
import { getThemeColors } from '@groxigo/tokens';
import { useColorScheme } from 'react-native';

function MyComponent() {
  const colorScheme = useColorScheme();
  const theme = getThemeColors(colorScheme === 'dark');

  return (
    <View style={{ backgroundColor: theme.semantic.surface.primary }}>
      <Text style={{ color: theme.semantic.text.primary }}>
        Hello World
      </Text>
    </View>
  );
}

Three-Tier Architecture

/* Tier 1: Primitives - Raw color values */
--color-blue-600: #2563eb;

/* Tier 2: Semantic - Meaningful names */
--brand-primary: var(--color-blue-600);

/* Tier 3: Components - UI element tokens */
--button-primary-bg: var(--brand-primary);

/* Usage: Always use the highest abstraction level */
.button {
  background: var(--button-primary-bg); /* Correct */
  /* NOT: var(--color-blue-600) */
}
